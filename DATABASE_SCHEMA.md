# #Connect Database Schema

Complete database schema for the #Connect community management platform.

---

## 📊 Database Overview

**Database:** Firebase Firestore
**Structure:** Collection-based NoSQL
**Geographic Scope:** Communities within 5KM radius

---

## 🗂️ Collections Overview

```
┌─────────────────────────────────────────────────────────┐
│                    FIRESTORE DATABASE                   │
├─────────────────────────────────────────────────────────┤
│ 1. users          - User profiles & authentication      │
│ 2. communities    - Geographic communities (5km radius)  │
│ 3. posts          - News & updates by admins            │
│ 4. polls          - Community polls                     │
│ 5. alerts         - Emergency alerts & notifications    │
│ 6. bills          - Utility & custom bills              │
│ 7. payments       - Payment transactions                │
│ 8. bill_payments  - Bill payment records                │
│ 9. polls_votes    - Poll voting records                 │
│ 10. notifications - User notifications                  │
└─────────────────────────────────────────────────────────┘
```

## 1. users Collection

Stores user information and roles.

### Document Structure

```typescript
{
  id: string                    // Auto-generated by Firebase Auth
  phone: string                 // +919876543210
  fullName: string              // "Shyam Dasari"
  email?: string                // Optional email
  role: "user" | "admin" | "super_admin"
  communityId: string           // ID of community user belongs to
  location: {
    latitude: number            // 17.385044
    longitude: number           // 78.486671
    address: string             // "Hyderabad, Telangana"
  }
  isVerified: boolean           // true if phone verified
  isActive: boolean             // true if account active
  avatar?: string               // Profile picture URL
  createdAt: Timestamp          // When user registered
  updatedAt: Timestamp          // Last profile update
  lastLoginAt: Timestamp        // Last login timestamp
  fcmToken?: string             // Push notification token
  preferences: {
    language: string            // "en" | "te" | "hi"
    notifications: {
      alerts: boolean           // Receive emergency alerts
      bills: boolean            // Receive bill reminders
      updates: boolean          // Receive community updates
      polls: boolean            // Receive poll notifications
    }
  }
  metadata: {
    deviceInfo?: string         // Device type
    ipAddress?: string          // Registration IP
  }
}
```

### Indexes

```typescript
// Compound indexes
users-community-role: [communityId, role]
users-phone: [phoneNumber]
users-location: [location.latitude, location.longitude]
```

## 2. communities Collection

Defines geographic communities (5KM radius areas).

### Document Structure

```typescript
{
  id: string                    // Auto-generated
  name: string                  // "Hyderabad - Hitech City"
  description: string           // "Community for Hitech City area"
  location: {
    center: {
      latitude: number          // 17.385044
      longitude: number         // 78.486671
    }
    radius: number              // 5000 (meters - 5KM)
    city: string                // "Hyderabad"
    state: string               // "Telangana"
    country: string             // "India"
    pincode: string             // "500081"
  }
  boundaries: {
    // Polygon defining community boundary
    type: "Polygon"
    coordinates: number[][]     // [[lat, lng], [lat, lng], ...]
  }
  adminId: string               // Primary admin user ID
  stats: {
    totalUsers: number          // Total registered users
    activeUsers: number         // Active in last 30 days
    totalPosts: number          // Total posts created
    totalPolls: number          // Total polls created
  }
  settings: {
    isPublic: boolean           // Can anyone join?
    requiresApproval: boolean   // Admin approval needed
    allowGuests: boolean        // Non-members can view
  }
  createdAt: Timestamp
  updatedAt: Timestamp
}
```

### Subcollections

#### communities/{id}/admins
```typescript
{
  userId: string                // Admin user ID
  role: "primary" | "secondary" | "moderator"
  permissions: {
    manageUsers: boolean        // Add/remove users
    createPosts: boolean        // Create news/posts
    createPolls: boolean        // Create polls
    sendAlerts: boolean         // Send emergency alerts
    manageBills: boolean        // Create/manage bills
    viewPayments: boolean       // View payment history
  }
  assignedAt: Timestamp
  assignedBy: string            // User ID who assigned this role
}
```

## 🔒 Security Rules

### Firestore Security Rules

```javascript
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Helper functions
    function isAuthenticated() {
      return request.auth != null;
    }

    function isUser(userId) {
      return isAuthenticated() && request.auth.uid == userId;
    }

    function isAdmin(communityId) {
      return isAuthenticated() &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role in ['admin', 'super_admin'] &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.communityId == communityId;
    }

    function isSuperAdmin() {
      return isAuthenticated() &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'super_admin';
    }

    // Users collection
    match /users/{userId} {
      allow read: isUser(userId) || isSuperAdmin();
      allow write: isUser(userId) || isSuperAdmin();
    }

    // Communities collection
    match /communities/{communityId} {
      allow read: isAuthenticated();
      allow write: isSuperAdmin();
      match /admins/{adminId} {
        allow read: isAdmin(communityId) || isSuperAdmin();
        allow write: isSuperAdmin();
      }
    }

    // Posts collection
    match /posts/{postId} {
      allow read: isAuthenticated();
      allow create: isAdmin(resource.data.communityId);
      allow update: isAdmin(resource.data.communityId);
      allow delete: isSuperAdmin();
      match /comments/{commentId} {
        allow read: isAuthenticated();
        allow create: isAuthenticated();
        allow update: isUser(resource.data.userId);
      }
    }

    // Polls collection
    match /polls/{pollId} {
      allow read: isAuthenticated();
      allow create: isAdmin(resource.data.communityId);
      allow update, delete: isAdmin(resource.data.communityId);
      match /votes/{voteId} {
        allow create: isUser(resource.data.userId);
        allow read: isAdmin(resource.data.communityId) || isSuperAdmin();
      }
    }

    // Alerts collection
    match /alerts/{alertId} {
      allow read: isAuthenticated();
      allow create: isAdmin(resource.data.communityId);
      allow update, delete: isAdmin(resource.data.communityId);
    }

    // Bills collection
    match /bills/{billId} {
      allow read: isUser(resource.data.userId) || isAdmin(resource.data.communityId);
      allow create: isAdmin(resource.data.communityId);
      allow update: isAdmin(resource.data.communityId);
      allow delete: isSuperAdmin();
    }

    // Payments collection
    match /payments/{paymentId} {
      allow read: isUser(resource.data.userId) || isSuperAdmin();
      allow create: isAuthenticated(); // Via API
      allow update: isSuperAdmin();
    }

    // Notifications collection
    match /notifications/{notificationId} {
      allow read: isUser(resource.data.userId);
      allow write: isSuperAdmin();
    }
  }
}
```
